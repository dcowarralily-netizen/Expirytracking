function onScanSuccess(decodedText, decodedResult) { const now = Date.now(); if (decodedText === lastScannedCode && (now - lastScannedTime < 2200)) return; lastScannedCode = decodedText; lastScannedTime = now; playBeepTone(); const nameField = document.getElementById("prodName"); const statusDiv = document.getElementById("scannerStatus"); const resultBox = document.getElementById("scanResultBox"); const resName = document.getElementById("resName"); const resBarcode = document.getElementById("resBarcode"); const resCost = document.getElementById("resCost"); const resSell = document.getElementById("resSell"); const currentForm = document.getElementById("expiryForm"); let cleanScannedCode = decodedText.replace(/[^0-9a-zA-Z]/g, '').trim(); // 1. Try an exact match first let match = productDatabase[cleanScannedCode]; // 2. Fallback: If no exact match, extract the last 6 digits and search your database if (!match && cleanScannedCode.length >= 6) { let scannedSuffix = cleanScannedCode.slice(-6); // gets "140341" let fallbackKey = Object.keys(productDatabase).find(key => { return key.endsWith(scannedSuffix); }); if (fallbackKey) { match = productDatabase[fallbackKey]; } } if (match) { let costNum = parseFloat(match.cost || 0); let sellNum = parseFloat(match.sell || 0); nameField.value = match.name; resName.innerText = match.name; resBarcode.innerText = decodedText; resCost.innerText = costNum > 0 ? "$" + costNum.toFixed(2) : "$-"; resSell.innerText = sellNum > 0 ? "$" + sellNum.toFixed(2) : "$-"; resultBox.style.display = "block"; statusDiv.innerHTML = `✅ Item Identified via Deep Suffix Match: ${match.name}`; statusDiv.style.backgroundColor = "#dcfce7"; statusDiv.style.color = "#155724"; currentForm.dataset.scannedCost = match.cost; currentForm.dataset.scannedSell = match.sell; document.getElementById("prodExpiry").focus(); } else { nameField.value = `Unlisted Barcode (${decodedText})`; resName.innerText = "Unknown Product Asset"; resBarcode.innerText = decodedText; resCost.innerText = "N/A"; resSell.innerText = "N/A"; resultBox.style.display = "block"; statusDiv.innerHTML = `❓ Barcode unrecognized (${decodedText}). Name auto-populated.`; statusDiv.style.backgroundColor = "#fef3c7"; statusDiv.style.color = "#856404"; currentForm.dataset.scannedCost = "0"; currentForm.dataset.scannedSell = "0"; nameField.focus(); } }
